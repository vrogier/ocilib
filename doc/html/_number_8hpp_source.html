95inline ostring Number::ToString(const ostring& format) const
96{
97 if (!IsNull())
98 {
99 const size_t size = OCI_SIZE_BUFFER;
100
101 core::ManagedBuffer<otext> buffer(static_cast<size_t>(size + 1));
102
103 core::Check(OCI_NumberToText(*this, format.c_str(), static_cast<int>(size), buffer));
104
105 return core::MakeString(static_cast<const otext *>(buffer));
106 }
107
108 return OCI_STRING_NULL;
109}
