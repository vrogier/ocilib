32inline Environment::Environment(EnvironmentFlags mode, const ostring& libpath) : _guard(core::SynchronizationMode::Unsafe), _defaultStore(nullptr), _charMaxSize(0), _initialized(false)
33{
34 _mode = mode;
35
36 core::Check(OCI_Initialize(nullptr, libpath.c_str(), _mode.GetValues() | OCI_ENV_CONTEXT));
37 _initialized = true;
38
39 core::Check(OCI_SetUserData(nullptr ,this));
40
41 _guard.SetMode((_mode & Environment::Threaded) == Environment::Threaded ? core::SynchronizationMode::Safe : core::SynchronizationMode::Unsafe);
42 _callbacks.SetGuard(&_guard);
43 _handle.AcquireTransient
44 (
45 /* returned value IS NOT an OCI_Environment* but OCIEnv* direct handle
46 to be changed when C API we have public methods for OCI_Environment */
47 static_cast<OCI_Environment*>(const_cast<AnyPointer>(core::Check(OCI_HandleGetEnvironment()))),
48 nullptr
49 );
50
51 _charMaxSize = ComputeCharMaxSize(GetCharset());
52}
53
54inline Environment::~Environment()
55{
56 _guard.SetMode(core::SynchronizationMode::Unsafe);
57 _callbacks.SetGuard(nullptr);
58 _handle.Release();
59
60 if (_initialized)
61 {
62 core::Check(OCI_SetUserData(nullptr ,this));
63 core::Check(OCI_Cleanup());
64 }
65}

67inline void Environment::Initialize(EnvironmentFlags mode, const ostring& libpath)
68{
69 Environment* environment = static_cast<Environment*>(OCI_GetUserData(nullptr));
70 if (environment == nullptr)
71 {
72 // not tracking return value here as created environment will be deallocated by Environment::Cleanup
73 core::OnAllocate(new Environment(mode, libpath));
74 }
75}

77inline void Environment::Cleanup()

85#ifdef OCILIBPP_DEBUG_MEMORY_ENABLED

91}

204inline void Environment::HAHandler(OCI_Connection *pConnection, unsigned int source, unsigned int event, OCI_Timestamp *pTimestamp)
205{
206 const HAHandlerProc handler = GetUserCallback<HAHandlerProc>(GetEnvironmentHandle());
207
208 if (handler)
209 {
210 Connection connection(pConnection, nullptr, false);
211 Timestamp timestamp(pTimestamp, connection.GetHandle());
212
213 handler(connection,
214 HAEventSource(static_cast<HAEventSource::Type>(source)),
215 HAEventType (static_cast<HAEventType::Type> (event)),
216 timestamp);
217 }
218}

220inline unsigned int Environment::TAFHandler(OCI_Connection *pConnection, unsigned int type, unsigned int event)
221{
222 unsigned int res = OCI_FOC_OK;
223
224 const Connection::TAFHandlerProc handler = GetUserCallback<Connection::TAFHandlerProc>(core::Check(pConnection));
225
226 if (handler)
227 {
228 Connection connection(pConnection, nullptr, false);
229
230 res = handler(connection,
231 Connection::FailoverRequest( static_cast<Connection::FailoverRequest::Type> (type)),
232 Connection::FailoverEvent ( static_cast<Connection::FailoverEvent::Type> (event)));
233 }
234
235 return res;
236}
