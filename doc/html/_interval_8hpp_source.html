274inline ostring Interval::ToString(int leadingPrecision, int fractionPrecision) const
275{
276 if (!IsNull())
277 {
278 const size_t size = OCI_SIZE_BUFFER;
279
280 core::ManagedBuffer<otext> buffer(static_cast<size_t>(size + 1));
281
282 core::Check(OCI_IntervalToText(*this, leadingPrecision, fractionPrecision, static_cast<int>(size), buffer));
283
284 return core::MakeString(static_cast<const otext *>(buffer));
285 }
286
287 return OCI_STRING_NULL;
288}
