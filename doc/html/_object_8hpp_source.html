374inline ostring Object::ToString() const
375{
376 if (!IsNull())
377 {
378 unsigned int size{};
379
380 core::Check(OCI_ObjectToText(*this, &size, nullptr));
381
382 ostring result;
383
384 result.resize(static_cast<size_t>(size));
385
386 core::Check(OCI_ObjectToText(*this, &size, const_cast<otext*>(result.data())));
387
388 result.resize(static_cast<size_t>(size));
389
390 return result;
391 }
392
393 return OCI_STRING_NULL;
394}
